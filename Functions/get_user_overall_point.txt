CREATE OR REPLACE 
FUNCTION GET_USER_OVERALL_POINT (u_id IN NUMBER)
RETURN NUMBER
IS
total NUMBER;
tid NUMBER;

BEGIN
	-- routine body goes here, e.g.
	-- DBMS_OUTPUT.PUT_LINE('Navicat for Oracle');
	tid:=CURRENT_TOURNAMENT();
	total:=0;
	FOR R IN (SELECT "phase_id" FROM "phase"  WHERE "tournament_id" = tid)
		LOOP
			total:=total+GET_USER_PHASE_POINT(u_id, R."phase_id");
		END LOOP;
	RETURN total;
END;
