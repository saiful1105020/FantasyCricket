  MINVALUE 1
  MAXVALUE 999999999999999999999999999
  START WITH 1

  CACHE 20;


CREATE or replace TRIGGER user_tournament_auto_increment 
	before INSERT on "user_tournament" 
		for each ROW 
			BEGIN
				if :NEW."user_tournament_id" is null then 
						:NEW."user_tournament_id" := user_tournament_id_counter.nextval; 
				end if;
